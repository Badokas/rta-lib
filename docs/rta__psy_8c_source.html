80 static int
81 estimateCandidates(float *input, float *corrBuffer, double maxPeriod, double minPeriod, double noiseThreshold, rta_psy_candidate_t *candidates)
82 {
83  int numCandidates = 1;
84  int nCorr = (int)ceil(maxPeriod);
85  double x, xn;
86  double energy;
87  double diffLeft, diff, diffRight, sum;
88  int i;
89 
90  /* init absolute minimum */
91  candidates[0].period = maxPeriod;
92  candidates[0].normDiff = 1.0;
93 
94  /* auto-correlation */
95  yinAutocorr(input, corrBuffer, nCorr, nCorr);
96 
97  /* diff[0] */
98  x = input[0];
99  xn = input[nCorr];
100  energy = corrBuffer[0] + xn * xn - x * x;
101  diffLeft = 0.0;
102  diff = 0.0;
103  diffRight = corrBuffer[0] + energy - 2 * corrBuffer[1];
104  sum = 0.0;
105 
106  /* diff[1] */
107  x = input[1];
108  xn = input[1 + nCorr];
109  energy += xn * xn - x * x;
110  diffLeft = diff;
111  diff = diffRight;
112  diffRight = corrBuffer[0] + energy - 2 * corrBuffer[2];
113  sum = diff;
114 
115  /* minimum difference search */
116  for(i = 2; i < nCorr - 1; i++)
117  {
118  x = input[i];
119  xn = input[i + nCorr];
120  energy += xn * xn - x * x;
121  diffLeft = diff;
122  diff = diffRight;
123  diffRight = corrBuffer[0] + energy - 2 * corrBuffer[i + 1];
124  sum += diff;
125 
126  /* local minimum */
127  if(diff < diffLeft && diff < diffRight)
128  {
129  double a = diffLeft + diffRight - 2.0 * diff;
130  double b = 0.5 * (diffRight - diffLeft);
131  double m = diff - (b * b) / (2.0 * a);
132  double period = i - b / a;
133  double normDiff = i * m / sum;
134 
135  if(period >= minPeriod)
136  {
137  if(normDiff < 0.0)
138  normDiff = 0.0;
139 
140  /* store candidate */
141  if(normDiff <= noiseThreshold)
142  {
143  candidates[numCandidates].period = period;
144  candidates[numCandidates].normDiff = normDiff;
145  numCandidates++;
146  }
147 
148  /* get absolute minimum */
149  if(normDiff < candidates[0].normDiff)
150  {
151  candidates[0].period = period;
152  candidates[0].normDiff = normDiff;
153  }
154  }
155 
156  if(numCandidates >= RTA_PSY_MAX_CANDIDATES)
157  break;
158  }
159  }
160 
161  return numCandidates;
162 }

165 traceBackAndForth(rta_psy_ana_t *self, rta_psy_tracking_state_t *trackingStates, int trackingIndex, rta_psy_candidate_t *candidate, int candidateIndex)
166 {
167  rta_psy_tracking_state_t *backwardState = trackingStates + (trackingIndex + RTA_PSY_NUM_TRACKING_STATES - 1) % RTA_PSY_NUM_TRACKING_STATES;
168  double nearestDiff = DBL_MAX;
169  int backwardIndex = 0;
170  int i;
171 
172  /* get closest pitch in last state */
173  for(i = 1; i < backwardState->numCandidates; i++)

176  double diff = fabs(pitch - backwardState->candidates[i].pitch);
177

180

182  if(diff > nearestDiff)
183  break;
184

186  backwardIndex = i;
187  }
188

208 static int
209 getBestCandidate(rta_psy_ana_t *self, rta_psy_tracking_state_t *trackingStates, int trackingIndex)
210 {
211  rta_psy_tracking_state_t *state = trackingStates + trackingIndex;
212  double yinThreshold = self->yinThreshold;
213  int i;
214 
215  yinThreshold += state->candidates[0].normDiff;
216 
217  for(i = 1; i < state->numCandidates; i++)
218  {
219  rta_psy_candidate_t *candidate = state->candidates + i;
220  double normDiff = candidate->normDiff;
221 
222  /* finish backward tracking */
223  if(candidate->forwardIndex > 0)
224  {
225  rta_psy_tracking_state_t *forwardState = trackingStates + (trackingIndex + RTA_PSY_NUM_TRACKING_STATES + 1) % RTA_PSY_NUM_TRACKING_STATES;
226  rta_psy_candidate_t *forwardCandidate = forwardState->candidates + candidate->forwardIndex;
227 
228  candidate->backward += 1.0 * forwardCandidate->backward;
229  candidate->backward *= 0.5;
230 
231  if(candidate->backward < normDiff)
232  normDiff = candidate->backward;
233  }
234 
235  if(candidate->forward < normDiff)
236  normDiff = candidate->forward;
237 
238  /* classic yin: look for first minimum under threshold */
239  if(normDiff < yinThreshold)
240  return i;
241  }
242 
243  return 0;

258

260  state->energy = sqrt(self->corrBuffer[0] * norm);
261  state->ac1 = self->corrBuffer[1] * norm;
262 
263  /* bias threshold by absolute minimum */
264  yinThreshold += state->candidates[0].normDiff;
265 
266  /* get yin index */
267  for(i = 1; i < state->numCandidates; i++)
268  {
269  rta_psy_candidate_t *candidate = state->candidates + i;
270  double normDiff = candidate->normDiff;
271 
272  /* init tracing */
273  candidate->pitch = DBL_MAX;
274  candidate->forward = 0.0;
275  candidate->forwardIndex = 0;
276  candidate->backward = 0.0;
277  candidate->backwardIndex = 0;
278

280  traceBackAndForth(self, trackingStates, trackingIndex, state->candidates + i, i);
281

506 rta_psy_calculate_input_vector(rta_psy_ana_t *self, float *in, int vectorSize, int vectorStride)
507 {
508  int downVectorSize = vectorSize >> self->downSamplingExp;
509  float *inputBuffer = self->inputBuffer + self->inputFill;
510  double outputTime = self->outputTime;
511  int maxTime;
512  int i, j;
513 
514  if(downVectorSize > 0)
515  {
516  switch(self->downSamplingExp)
517  {
518  case 3:
519  {
520  for(i = 0, j = 0; i < downVectorSize; i++, j += 8 * vectorStride)
521  inputBuffer[i] = 0.125 * (in[j] + in[j + vectorStride] +
522  in[j + 2 * vectorStride] + in[j + 3 * vectorStride] +
523  in[j + 4 * vectorStride] + in[j + 5 * vectorStride] +
524  in[j + 6 * vectorStride] + in[j + 7 * vectorStride]);
525  break;
526  }
527 
528  case 2:
529  {
530  for(i = 0, j = 0; i < downVectorSize; i++, j += 4 * vectorStride)
531  inputBuffer[i] = 0.25 * (in[j] + in[j + vectorStride] +
532  in[j + 2 * vectorStride] + in[j + 3] * vectorStride);
533 
534  break;
535  }
536 
537  case 1:
538  {
539  for(i = 0, j = 0; i < downVectorSize; i++, j += 2 * vectorStride)
540  inputBuffer[i] = 0.5 * (in[j] + in[j + vectorStride]);
541 
542  break;
543  }
544 
545  default:
546  {
547  for(i = 0, j = 0; i < downVectorSize; i++, j += vectorStride)
548  inputBuffer[i] = in[j];
549  }
550  }
551  }
552  else
553  {
554  float sum = 0.0;
555 
556  for(i = 0; i < vectorSize * vectorStride; i += vectorStride)
557  sum += in[i];
558 
559  inputBuffer[0] = sum / (float)vectorSize;
560  }

572
