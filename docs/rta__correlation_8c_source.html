47 /* Fast, unbiased by nature, recommended if (c_size / filter_size > 20) */
48 /* Requirement: (a_size, b_size) >= c_size + filter_size */
49 /* Warning: for VecLib, a_size is required to be aligned on a multiple */
50 /* of 4, that is */
51 /* a_size >= 'c_size' + 4*(('filter_size'+ 4 - 1)/4) */
52 void rta_correlation_fast(
53  rta_real_t * correlation, const unsigned int c_size,
54  const rta_real_t * input_vector_a,
55  const rta_real_t * input_vector_b,
56  const unsigned int filter_size)
57 {
58 #if defined(RTA_USE_VECLIB)
59  if(filter_size <= 2044 && filter_size + c_size >=12)
60  {
61 #if (RTA_REAL_TYPE == RTA_FLOAT_TYPE)
62  vDSP_conv(input_vector_a, 1, input_vector_b, 1, correlation, 1,
63  c_size, filter_size);
64 #elif (RTA_REAL_TYPE == RTA_DOUBLE_TYPE)
65  vDSP_convD(input_vector_a, 1, input_vector_b, 1, correlation, 1,
66  c_size, filter_size);
67 #endif
68  }
69  else
70  {
71 #endif /* RTA_USE_VECLIB */
72 
73 /* Base algorithm */
74  int c,f;
75  for(c=0; c<c_size; c++)
76  {
77  correlation[c] = 0.0;
78  for(f=0; f<filter_size; f++)
79  {
80  correlation[c] += input_vector_a[f+c] * input_vector_b[f];
81  }
82  } /* end of base algorithm */
83 
84 #if defined(RTA_USE_VECLIB)
85  }
86 #endif
87  return;
