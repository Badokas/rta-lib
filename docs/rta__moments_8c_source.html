47 /* 1st order weighted moment over indexes: weighted mean, centroid */
48 rta_real_t rta_weighted_moment_1_indexes(
49  rta_real_t * input_sum,
50  const rta_real_t * input, const unsigned int input_size)
51 {
52  unsigned int i;
53  rta_real_t centroid = 0.;
54  *input_sum = 0.;
55 
56  for(i=0; i<input_size; i++)
57  {
58  centroid += input[i] * i;
59  *input_sum += input[i];
60  }
61 
62  if(*input_sum > 0.)
63  {
64  centroid /= *input_sum;
65  }
66  else
67  {
68  /* flat and null input => centroid is the middle */
69  centroid = (input_size - 1) * 0.5;
70  }
71 
72  return centroid;
73 }
