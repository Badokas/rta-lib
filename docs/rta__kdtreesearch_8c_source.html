132 rta_real_t rta_euclidean_distance (rta_real_t* v1, int stride1,
133  rta_real_t* v2, int dim,
134  rta_bpf_t *distfunc[])
135 {
136  int i, i1;
137  rta_real_t sum = 0;
138 
139  for (i = 0, i1 = 0; i < dim; i++, i1 += stride1)
140  {
141 #if USE_DISTFUNC // uses rta_bpf_t, (data-compatible to FTM bpfunc_t)
142  rta_real_t diff = v2[i] - v1[i1];
143  rta_bpf_t *dfun = distfunc[i];
144  if (dfun)
145  diff = rta_bpf_get_interpolated(dfun, diff);
146 #else
147  rta_real_t diff = v2[i] - v1[i1];
148 #endif /* USE_DISTFUNC */
149  sum += diff * diff;
150  }
151 
152  return sum;
153 }

156 rta_real_t rta_weighted_euclidean_distance (rta_real_t* v1,
157  rta_real_t* v2,
158  rta_real_t *sigma, int ndim,
159  rta_bpf_t *distfunc[])
160 {
161  int i;
162  rta_real_t sum = 0, sqrtsum = 0;
163 
164  for (i = 0; i < ndim; i++)
165  if (sigma[i] > 0)
166  {
167 #if USE_DISTFUNC // uses rta_bpf_t, (data-compatible to FTM bpfunc_t)
168  rta_real_t diff = v2[i] - v1[i];
169  rta_bpf_t *dfun = distfunc[i];
170  if (dfun)
171  diff = rta_bpf_get_interpolated(dfun, diff);
172  diff /= sigma[i];
173 #else
174  rta_real_t diff = (v2[i] - v1[i]) / sigma[i];
175 #endif /* USE_DISTFUNC */
176  sum += diff * diff;
177

179  i, v2[i], v1[i], diff, sum, sqrt(sum));
180  }
181  sqrtsum = sqrt(sum);
182  return sqrtsum;
183 }

213  dist[K] = distance of the Kth nearest neighbour
214  return: actual number of found neighbours */
215 int kdtree_search_knn (kdtree_t *t, rta_real_t* vector, int stride,
216  int k, const rta_real_t r, int use_sigma,
217  /* out */ kdtree_object_t *indx, rta_real_t *dist)
218 {
219  int kmax = 0; /* index of current kth neighbour */
220  int leaves_start = t->ninner; /* first leaf node */
221  rta_real_t sentinel = (r == 0 ? MAX_FLOAT : r);
222  rta_real_t *sigmaptr = t->sigma;
223  rta_real_t dxx; /* distance between 2 vectors */
224  int i; /* current processed vector */
225 
226  kdtree_stack_t *s = &t->stack;
227  kdtree_stack_elem_t cur; /* current (node, dist) couple */
228 
229  if (t->ndatatot == 0)
230  return 0;
231 
232  if (k < 1)
233  k = 1;
234 
235  // Init distances
236  for (i = 0; i < k; i++)
237  dist[i] = sentinel;
238 
239  // Init Search Stack
240  stack_clear(s);
241  stack_push(s, 0, 0);
242 
243  while (!stack_empty(s))
244  {
245 #if DEBUG_KDTREESEARCH >= 2
246  kdtree_stack_display(s);
247 #endif
248 #if KDTREE_PROFILE_SEARCH
249  if (s->size > t->profile.maxstack)
250  t->profile.maxstack = s->size;
251 #endif
252  stack_pop(s, &cur);
253 
254  if (cur.dist <= dist[kmax]) // elimination rule
255  {
256  if (cur.node >= leaves_start)
257  { /* leaf node: search through vectors linearly */
258  int istart = t->nodes[cur.node].startind;
259  int iend = t->nodes[cur.node].endind;
260  int i;
261 
262 #if DEBUG_KDTREESEARCH
263  rta_post("Leaf node p = %d cur.dist %f\n", cur.node, cur.dist);
264 #endif
265  for (i = istart; i <= iend; i++)
266  {
267  if (use_sigma)
268  dxx = rta_weighted_euclidean_distance_stride(vector, stride,
269  kdtree_get_vector(t, i), sigmaptr, t->ndim, t->dfun);
270  else
271  dxx = rta_euclidean_distance(vector, stride,
272  kdtree_get_vector(t, i), t->ndim, t->dfun);
273 #if KDTREE_PROFILE_SEARCH
274  t->profile.v2v++;
275 #endif
276 #if DEBUG_KDTREESEARCH
277  rta_post(" distance = %f between vector %d (elem %d, %d) ", dxx, i, t->dataindex[i].base, t->dataindex[i].index);
278  vec_post(kdtree_get_vector(t, i), 1, t->ndim, " and x ");
279  vec_post(vector, stride, t->ndim, "\n");
280 #endif
281  if (dxx <= dist[kmax])
282  { /* return original index in data and distance */
283  if (k == 1)
284  {
285  indx[kmax] = t->dataindex[i];
286  dist[kmax] = dxx;
287  }
288  else if (t->sort)
289  {
290  int pos = kmax; /* where to insert */
291 
292  if (kmax < k - 1)
293  { /* first move or override */
294  dist[kmax + 1] = dist[kmax];
295  indx[kmax + 1] = indx[kmax];
296  kmax++;
297  }
298 
299  /* insert into sorted list of distance */
300  while (pos > 0 && dxx < dist[pos - 1])
301  { /* move up */
302  dist[pos] = dist[pos - 1];
303  indx[pos] = indx[pos - 1];
304  pos--;
305  }
306 
307  indx[pos] = t->dataindex[i];
308  dist[pos] = dxx;
309  }
310  else
311  {
312  indx[kmax] = t->dataindex[i];
313  dist[kmax] = dxx;
314  kmax = maxArr(dist, k);
315  }
316  }
317  }
318  }
319  else
320  { // branched node
321  rta_real_t d;
322 
323  if (use_sigma)
324  d = distV2N_weighted(t, vector, stride, sigmaptr, cur.node);
325  else
326  d = distV2N_stride(t, vector, stride, cur.node);
327 
328 #if DEBUG_KDTREESEARCH
329  rta_post("Inner node %d d %f cur.dist %f --> push max %f\n",
330  cur.node, d, cur.dist, MAX(cur.dist, d*d));
331 #endif
332  if (d < 0)
333  {
334  stack_push(s, 2*cur.node+2, MAX(cur.dist, d*d));
335  stack_push(s, 2*cur.node+1, cur.dist);
336  }
337  else
338  {
339  stack_push(s, 2*cur.node+1, MAX(cur.dist, d*d));
340  stack_push(s, 2*cur.node+2, cur.dist);
341  }
342  }
343  }
344 #if DEBUG_KDTREESEARCH
345  else /* node can be eliminated from search */
346  {
347  rta_post("eliminate node %d (size %d): cur.dist %f > dist[kmax=%d] = %f\n",
348  cur.node, t->nodes[cur.node].size, cur.dist, kmax, dist[kmax]);
349  }
350 #endif
351 
352  }
353 #if KDTREE_PROFILE_SEARCH
354  t->profile.searches++;
355  t->profile.neighbours += kmax + 1;
356 #endif
357 #if DEBUG_KDTREESEARCH
358  rta_post("kdtree_search found %d vectors < radius %f\n",
359  kmax + (dist[kmax] < sentinel), r);
360 #endif
361 
362  /* return actual number of found neighbours, can be less than k,
363  then kmax is the index of the next one to find */
364  return kmax + (dist[kmax] < sentinel);
365 }
