249 int rta_kdtree_set_data (rta_kdtree_t *self, int nblocks, rta_real_t **data,
250  rta_kdtree_object_t *index, int *m, int n)
251 {
252  int maxheight, givenheight, height;
253  int i, j = 0, k;
254 
255  self->data = data;
256  self->nblocks = nblocks;
257  self->ndata = m;
258  self->ndim = n;
259 
260  self->ndatatot = 0;
261 
262  for (i = 0; i < nblocks; i++)
263  self->ndatatot += self->ndata[i];
264 
265  maxheight = floor(log2(self->ndatatot));
266  givenheight = self->givenheight;
267  height = givenheight > 0 ? givenheight : maxheight + givenheight;
268 
269  /* clip height */
270  if (height > maxheight)
271  height = maxheight;
272 
273  if (height < 1)
274  height = 1; /* minimum: just one node, linear search */
275 
276  self->maxheight = maxheight;
277  self->height = height;
278 
279  /* init num nodes */
280  self->nnodes = pow2(height) - 1;
281  self->ninner = pow2(height - 1) - 1;
282 
283  /* init original index list */
284  rta_auto_alloc(self->dataindex, index, self->ndatatot);
285 
286  if (index == NULL) /* no indices given, create them ourselves; else: use indices from outside */
287  for (k = 0; k < nblocks; k++)
288  for (i = 0; i < m[k]; i++, j++)
289  {
290  self->dataindex[j].base = k;
291  self->dataindex[j].index = i;
292  }
293 
294  /* init search stack size according to tree height
295  (with heuristic margin of 4 times) */
296  rta_kdtree_stack_grow(&self->stack, self->height * 4);
297 
298  return self->nnodes;
299 }
