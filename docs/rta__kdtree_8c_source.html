246 int kdtree_set_data (kdtree_t *self, int nblocks, rta_real_t **data, kdtree_object_t *index,
247  int *m, int n)
248 {
249  int maxheight, givenheight, height;
250  int i, j = 0, k;
251 
252  self->data = data;
253  self->nblocks = nblocks;
254  self->ndata = m;
255  self->ndim = n;
256 
257  self->ndatatot = 0;
258  for (i = 0; i < nblocks; i++)
259  self->ndatatot += self->ndata[i];
260 
261  maxheight = floor(log2(self->ndatatot));
262  givenheight = self->givenheight;
263  height = givenheight > 0 ? givenheight : maxheight + givenheight;
264 
265  /* clip height */
266  if (height > maxheight)
267  height = maxheight;
268  if (height < 1)
269  height = 1; /* minimum: just one node, linear search */
270 
271  self->maxheight = maxheight;
272  self->height = height;
273 
274  /* init num nodes */
275  self->nnodes = pow2(height) - 1;
276  self->ninner = pow2(height - 1) - 1;
277 
278  /* init original index list */
279  auto_alloc(self->dataindex, index, self->ndatatot);
280  if (index == NULL) /* no indices given, create them ourselves; else: use indices from outside */
281  for (k = 0; k < nblocks; k++)
282  for (i = 0; i < m[k]; i++, j++)
283  {
284  self->dataindex[j].base = k;
285  self->dataindex[j].index = i;
286  }
287 
288  /* init search stack size according to tree height
289  (with heuristic margin of 4 times) */
290  kdtree_stack_grow(&self->stack, self->height * 4);
291 
292  return self->nnodes;
293 }
